🔌 Connexion Supabase

Non configuré
Configuration en 4 étapes :
1. Allez sur supabase.com → créez un compte gratuit → nouveau projet "fastlivr"
2. Dans votre projet → SQL Editor → exécutez le SQL ci-dessous
3. Dans Settings → API → copiez Project URL et la clé anon public
4. Collez-les ici et cliquez Se connecter
CREATE TABLE IF NOT EXISTS companies ( id TEXT PRIMARY KEY, nom TEXT NOT NULL DEFAULT '', statut TEXT DEFAULT 'Prospect', responsable TEXT DEFAULT '', tel TEXT DEFAULT '', email TEXT DEFAULT '', commission TEXT DEFAULT '5', paiement TEXT DEFAULT 'Mobile Money', vehicule TEXT DEFAULT '', notes TEXT DEFAULT '', gares JSONB DEFAULT '[]', created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE companies ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "allow_all" ON companies; CREATE POLICY "allow_all" ON companies FOR ALL USING (true) WITH CHECK (true);
